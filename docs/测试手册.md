## ITSM Webhook测试

### 1. 防火墙策略Webhook测试

**测试脚本** `test_itsm_webhook.ps1`:

```powershell
$url = "http://localhost:8000/api/v1/itsm/webhook/firewall-policy"

$payload = @{
    ticket_id = "TEST001"
    ticket_title = "测试团队申请数据库访问防火墙策略开通"
    service_catalog = "安全-防火墙策略开通"
    requester = "test.user"
    requester_dept = "测试部"
    assignee = "admin"
    priority = "P2"
    due_date = "2024-05-21"
    policy_file = @{
        url = "C:\Users\wangd\PycharmProjects\PythonProject\netops-agent\tools\firewall-policy\policies.xlsx"
        filename = "policies.xlsx"
    }
    callback_url = "http://localhost:8000/api/v1/itsm/webhook/callback"
    callback_headers = @{
        "X-API-Key" = "itsm_callback_key_123"
    }
}

$headers = @{
    "Content-Type" = "application/json"
}

$response = Invoke-RestMethod -Uri $url -Method Post -Headers $headers -Body ($payload | ConvertTo-Json -Depth 10)
Write-Host "响应状态码: 202"
Write-Host "Celery Task ID: $($response.celery_task_id)"

# 等待任务执行
Start-Sleep -Seconds 5

# 查询任务状态
$statusUrl = "http://localhost:8000/api/v1/tasks/" + $response.celery_task_id
$statusResponse = Invoke-RestMethod -Uri $statusUrl -Method Get
Write-Host "任务状态: $($statusResponse.status)"
Write-Host "任务结果: $($statusResponse.result)"
```

**执行测试**:

## 性能测试

### 1. 并发任务测试

```python
import requests
import concurrent.futures
import time

def submit_task(i):
    url = "http://localhost:8000/api/v1/itsm/webhook/firewall-policy"
    payload = {
        "ticket_id": f"PERF{i:03d}",
        "ticket_title": f"性能测试{i}",
        "service_catalog": "安全-防火墙策略开通",
        "requester": "perf.user",
        "policy_file": {
            "url": "C:\\Users\\wangd\\PycharmProjects\\PythonProject\\netops-agent\\tools\\firewall-policy\\policies.xlsx",
            "filename": "policies.xlsx"
        },
        "callback_url": "http://localhost:8000/api/v1/itsm/webhook/callback"
    }
    start = time.time()
    response = requests.post(url, json=payload)
    elapsed = time.time() - start
    return {
        "index": i,
        "status_code": response.status_code,
        "elapsed": elapsed,
        "task_id": response.json().get("celery_task_id")
    }

# 提交10个并发任务
with concurrent.futures.ThreadPoolExecutor(max_workers=5) as executor:
    results = list(executor.map(submit_task, range(10)))

# 打印结果
for r in results:
    print(f"任务{r['index']}: 状态码={r['status_code']}, 耗时={r['elapsed']:.2f}秒")
```
